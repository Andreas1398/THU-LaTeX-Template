% Generate Signature line(s)
\newcommand{\getSignature}[1]{
    \ifdefempty{#1} % if parameter #1 is empty
    % then:
    {}
    % else:
    {
        \begin{minipage}{0.5\textwidth}\vspace{1.5cm}\rule{0.7\textwidth}{0.2mm}\newline#1\end{minipage}
    }
}

\newcommand{\signature}{
    \par\vspace*{0.8cm}\locationAndDate\newline\getSignature{\documentAuthor}\getSignature{\secondAuthor}\getSignature{\thirdAuthor}\getSignature{\fourthAuthor}\getSignature{\fifthAuthor}
}
