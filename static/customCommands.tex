\newcommand{\ifTrue}[3]
{
    \ifdefstring{#1}{true} % if parameter #1 is true
    % then:
    {#2}
    % else:
    {#3}
}

\newcommand{\ifFalse}[3]
{
    \ifdefstring{#1}{false} % if parameter #1 is true
    % then:
    {#2}
    % else:
    {#3}
}
